## Turn this into a list comprehension and make it more efficient!


linl = [[1,3,5,7,9],[2,4,6,8,10],[2,3,5,7,9,11]]
new = []
for i in linil:
    for j in i:
        new.append(j)
newnew = []
for n in new:
    if n not in newnew:
        newnew.append(n)
